    <script>
        const firebaseConfig = {
            apiKey: "AIzaSyDWQG53tP2zKILTwPSJQpiVzFNyvYLxLqw",
            authDomain: "ricebowlmonst.firebaseapp.com",
            databaseURL: "https://ricebowlmonst-default-rtdb.firebaseio.com",

        if (!firebase.apps.length) firebase.initializeApp(firebaseConfig);
        const db = firebase.database();

        let allProducts = [];
        let cart = {}; // { productId: { ...product, qty: 1 } }

        // Load Products
        db.ref('products').on('value', snap => {
            const val = snap.val();
            allProducts = [];
            if(val) {
                Object.keys(val).forEach(key => {
                    if(val[key].available !== false && val[key].available !== 'false') {
                        allProducts.push({ id: key, ...val[key] });
                    }
                });
            }
            renderMenu('all');
        });

        function renderMenu(category) {
            const container = document.getElementById('menu-container');
            container.innerHTML = '';
            
            const filtered = category === 'all' 
                ? allProducts 
                : allProducts.filter(p => (p.category || 'LAIN-LAIN') === category);

            if(filtered.length === 0) {
                container.innerHTML = '<div style="grid-column:1/-1; text-align:center; padding:40px; color:#9ca3af;">Menu tidak ditemukan.</div>';
                return;
            }

            filtered.forEach(p => {
                const div = document.createElement('div');
                div.className = 'product-card';
                div.innerHTML = `
                    <img src="${p.image || 'https://via.placeholder.com/150'}" class="prod-img" loading="lazy">
                    <div class="prod-info">
                        <div class="prod-name">${p.name}</div>
                        <div class="prod-price">Rp ${parseInt(p.price).toLocaleString('id-ID')}</div>
                        <button class="add-btn" onclick="addToCart('${p.id}')">TAMBAH</button>
                    </div>
                `;
                container.appendChild(div);
            });
        }

        window.filterMenu = function(cat, btn) {
            document.querySelectorAll('.cat-btn').forEach(b => b.classList.remove('active'));
            btn.classList.add('active');
            renderMenu(cat);
        }

        window.addToCart = function(id) {
            const prod = allProducts.find(p => p.id === id);
            if(!prod) return;

            if(cart[id]) {
                cart[id].qty++;
            } else {
                cart[id] = { ...prod, qty: 1 };
            }
            updateCartUI();
        }

        function updateCartUI() {
            const items = Object.values(cart);
            const totalQty = items.reduce((sum, item) => sum + item.qty, 0);
            const totalPrice = items.reduce((sum, item) => sum + (item.price * item.qty), 0);

            document.getElementById('cart-count').textContent = `${totalQty} Item`;
            document.getElementById('cart-total').textContent = `Rp ${totalPrice.toLocaleString('id-ID')}`;
            
            const bar = document.getElementById('cart-bar');
            if(totalQty > 0) bar.classList.add('show');
            else bar.classList.remove('show');
        }

        window.openCheckout = function() {
            const modal = document.getElementById('checkout-modal');
            const list = document.getElementById('checkout-items');
            list.innerHTML = '';
            
            Object.values(cart).forEach(item => {
                list.innerHTML += `
                    <div class="cart-item">
                        <span>${item.name} x${item.qty}</span>
                        <strong>Rp ${(item.price * item.qty).toLocaleString('id-ID')}</strong>
                    </div>
                `;
            });
            modal.classList.add('show');
        }

        window.closeCheckout = function() {
            document.getElementById('checkout-modal').classList.remove('show');
        }

        window.toggleTableInput = function() {
            const type = document.getElementById('order-type').value;
            document.getElementById('table-input-group').style.display = type === 'Dine-in' ? 'block' : 'none';
        }

        document.getElementById('order-form').addEventListener('submit', (e) => {
            e.preventDefault();
            const outlet = document.getElementById('outlet-select').value;
            const name = document.getElementById('cust-name').value;
            const type = document.getElementById('order-type').value;
            const table = document.getElementById('table-num').value;

            const orderData = {
                date: firebase.database.ServerValue.TIMESTAMP,
                status: 'Baru',
                payment: { method: 'Tunai', total: Object.values(cart).reduce((s, i) => s + (i.price * i.qty), 0) },
                details: { type: type, tableNumber: table || '-', customerName: name },
                items: Object.values(cart).map(i => ({ name: i.name, price: i.price, quantity: i.qty }))
            };

            db.ref(`orders/${outlet}`).push(orderData).then(() => {
                alert('Pesanan berhasil dikirim! Mohon tunggu konfirmasi kasir.');
                cart = {};
                updateCartUI();
                closeCheckout();
                document.getElementById('order-form').reset();
            });
        });

        // [BARU] Logic AI Chat
        function toggleChat() {
            const modal = document.getElementById('ai-chat-modal');
            const btn = document.getElementById('ai-chat-btn');
            
            if(modal.classList.contains('show')) {
                modal.classList.remove('show');
                btn.style.display = 'flex';
            } else {
                modal.classList.add('show');
                btn.style.display = 'none';
                document.getElementById('ai-input').focus();
            }
        }

        function handleChatInput(e) {
            if(e.key === 'Enter') sendUserMessage();
        }

        function sendUserMessage() {
            const input = document.getElementById('ai-input');
            const text = input.value.trim();
            if(!text) return;

            addMessage(text, 'user');
            input.value = '';

            // Efek mengetik
            setTimeout(() => {
                const response = getAIResponse(text);
                addMessage(response, 'bot');
            }, 600);
        }

        function addMessage(text, sender) {
            const container = document.getElementById('ai-chat-messages');
            const div = document.createElement('div');
            div.className = `ai-message ${sender}`;
            div.innerHTML = text; // Allow HTML for line breaks
            container.appendChild(div);
            container.scrollTop = container.scrollHeight;
        }

        function getAIResponse(input) {
            input = input.toLowerCase();
            
            if(input.includes('menu') || input.includes('makanan') || input.includes('lapar')) return "Kami punya aneka Rice Bowl lezat! 🍛<br><b>Best Seller:</b> Chicken Katsu Curry & Beef Teriyaki.<br>Mau pesan yang mana?";
            if(input.includes('minum') || input.includes('haus')) return "Segarkan harimu dengan <b>Es Teh Tarik</b> atau <b>Lemon Tea</b> kami! 🥤";
            if(input.includes('harga') || input.includes('mahal')) return "Tenang kak, harga mulai <b>Rp 15.000</b> saja sudah kenyang! Cek menu di atas ya.";
            if(input.includes('promo') || input.includes('diskon')) return "Cek tab <b>'Diskon'</b> di menu bawah ya! Sering ada promo Happy Hour lho! 🏷️";
            if(input.includes('buka') || input.includes('jam')) return "Kami buka setiap hari jam <b>10.00 - 22.00 WIB</b>. Ditunggu kedatangannya! 🕙";
            if(input.includes('lokasi') || input.includes('alamat')) return "Kami ada di Sidoarjo dan Surabaya. Klik menu <b>'Lokasi'</b> di bawah untuk peta lengkapnya! 📍";
            if(input.includes('halo') || input.includes('hi')) return "Halo Monster Lovers! 👋 Ada yang bisa saya bantu? Tanya menu atau promo yuk!";
            if(input.includes('makasih') || input.includes('thanks')) return "Sama-sama! Selamat makan enak! 😋";
            
            return "Maaf, saya Monster Bot 🤖. Saya bisa bantu info seputar <b>Menu, Promo, Jam Buka, atau Lokasi</b>. Coba tanya itu ya!";
        }
